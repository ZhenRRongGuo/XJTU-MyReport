\section{报告说明}

这是一份关于模板\hndm{MyReport.cls}的使用说明。本模板基于\LaTeX{}，请使用XeLaTeX进行编译。本模板在导言区通过命令\hndm{\documentclass{MyReport}}使用。

模板提供了4个必选参数，通过命令\hndm{\titlename{}}设置报告名称，\hndm{\stuname{}}设置姓名，\hndm{\stuID{}}设置学号，\hndm{\stuClass{}}设置班级。

模板提供了部分可选参数，具体说明及其使用方法如\cref{tab:可选参数说明}所示。

\begin{inparaenum}%[]
    \item 通过可选参数 \hndm{English}，支持全英文写作。

    \item 通过可选参数 \hndm{abstractType} 设置摘要类型，默认无摘要。

    如果设置了摘要，需要通过命令\hndm{\zhAbstractText{}}设置中文摘要内容，通过命令\hndm{\zhAbstractKey{}}设置中文摘要关键词，通过命令\hndm{\enAbstractText{}}设置英文摘要内容，通过命令\hndm{\enAbstractKey{}}设置英文摘要关键词。这些命令需要在导言区使用。

    \item 通过可选参数 \hndm{sectionType} 设置section排版类型，默认为连续排版。

    \item 通过可选参数\hndm{listType}设置抄录环境类型，默认为\hndm{minted}宏包提供环境，强烈建议保持默认。抄录环境的使用会在后文中详细说明。

    \item 默认情况下不会排版参考文献，可以通过可选参数\hndm{Reference={ref.bib}}设置参考文献文件地址，此时会排版参考文献，具体使用方法与效果见第 \ref{sec:参考文献的使用} 章。
\end{inparaenum}

\subsection{公式环境}
行内公式请使用美元符号\hndm{$···$}包裹，如下所示。

\begin{texcode}[]{}
    这是一个行内公式$y=kx+n$。
\end{texcode}

行间公式请使用\hndm{\equation}环境，模板会按章节进行编号。在\hndm{equation}环境中使用\hndm{\label{}}命令设置标签，从而可以交叉引用。模板建议为公式标签加入前缀eq，如下所示。

\begin{texcode}[]{}
    \begin{equation}
        \e^x =\sum\limits_{n=0}^{\infty} \frac{x^n}{n!}=1+x+\frac{x^2}{2!}+\cdots+\frac{x^n}{n!}+\cdots \qquad x\in(-\infty ,+\infty)
        \label{eq:公式示例，建议使用eq前缀}
    \end{equation}
\end{texcode}

对于多行公式，请使用\hndm{equation}环境嵌套\hndm{split}环境，这样整个公式使用一个编号。

\begin{equation}
    \begin{split}
        \int_{(L)}f(x,y) \diff s
         & =\int_\alpha^\beta f[\ x(t),y(t)\ ]\sqrt{[\ x'^2(t)+y'^2(t)\ ]} \diff t                   \\
         & =\int_a^b f(x,y(x))\sqrt{1+y'^2} \diff x                                                  \\
         & =\int_\alpha^\beta f(\rho \cos \theta,\rho \sin \theta)\sqrt{\rho^2+\rho'^2} \diff \theta
    \end{split}
    \label{eq:多行公式}
\end{equation}
\cref{eq:多行公式}的代码如下所示：
\begin{lstlisting}[]{latex}
    \begin{equation}
        \begin{split}
            \int_{(L)}f(x,y) \diff s
            &=\int_\alpha^\beta f[\ x(t),y(t)\ ]\sqrt{[\ x'^2(t)+y'^2(t)\ ]} \diff t \\
            &=\int_a^b f(x,y(x))\sqrt{1+y'^2} \diff x                                \\
            &=\int_\alpha^\beta f(\rho\cos\theta,\rho\sin\theta)\sqrt{\rho^2+\rho'^2}\diff\theta
        \end{split}
        \label{eq:多行公式}
    \end{equation}
\end{lstlisting}

\subsubsection{公式间距微调}
\begin{texcode}[]{}
    \begin{equation}
        f(x)=\sum_{n=-\infty}^{\infty} c_n \e^{\ii nx}
        \label{eq:f间距过小}
    \end{equation}
\end{texcode}

上述情况中，$f$与后面字符的间距过小，影响美观。为了解决这个问题，可以在字符$f$之前使用双引号\hndm{$"f$}改善间距。

\begin{texcode}[]{}
    \begin{equation}
        "f(x)=\sum_{n=-\infty}^{\infty} c_n \e^{\ii nx}
        \label{eq:f间距改善}
    \end{equation}
\end{texcode}

\begin{inparaenum}%[]

\end{inparaenum}

\section{算法与抄录环境}
\subsection{算法环境}
本模板使用\hndm{algorithm2e}宏包排版算法，具体使用方法见\hndm{algorithm2e}宏包手册。

\begin{algorithm}
    \caption{共轭梯度法}\label{alg:共轭梯度法}
    \KwIn{对称正定矩阵$\ve{A}$，常数向量$\ve{b}$，精度，最大迭代数}
    \KwOut{解向量 $\ve{x}$}
    $\ve{x}^0\leftarrow \textbf{0}$  \tcp*[r]{初始值}
    $\ve{r}\leftarrow \ve{b}-\ve{A}\ve{x}^0$ \;
    $\ve{d}\leftarrow \ve{r}$ \;
    误差 = $\|\ve{r} \| $\;
    \While{迭代次数 $<$ 最大迭代数}{
        $\alpha\leftarrow \frac{\ve{r}^{\rm T}\ve{d}}{\ve{d}^{\rm T}\ve{Ad}}$ \;
        $\ve{x}\leftarrow \ve{x}+\alpha\ve{d}$ \;
        $\ve{r}\leftarrow \ve{b}-\ve{Ax}$ \;
        误差 = $\|\ve{r} \| $\;
        \If{误差 $<$ 精度}{
            跳出循环 \;
        }
        $\beta\leftarrow -\frac{\ve{r}^{\rm T}\ve{Ad}}{\ve{d}^{\rm T}\ve{Ad}}$ \;
        $\ve{d}\leftarrow \ve{r}+\beta\ve{d}$ \;
    }
    返回解向量 $\ve{x}$ \;
\end{algorithm}

\subsection{抄录环境}

\begin{inparaenum}%[]
    \item 本模板使用\hndm{minted}宏包排版抄录。

    \item \hndm{minted}宏包需要Python环境的Pygments库，因为minted依赖于该模块提供超过300种语言的语法高亮，请确保电脑中安装了Python环境。

    \item \hndm{minted}宏包在编译时需要添加选项\hndm{-shell-escape}。（自行测试下来，不需要该选项似乎也能正常编译，如果一次编译不通过，连续编译两次即可。）

    \item \hndm{minted}宏包提供了\hndm{listing}浮动体环境，从而可以给抄录环境增加标题与标签，但这种环境不支持跨页，因此模板\textbf{不}建议使用。

    \item 模板基于\hndm{minted}宏包宏包定义了\hndm{lstlisting}环境，\textbf{建议}使用该环境排版，其使用方法如下所示。

    \begin{minted}{latex}

    def vec_add(x, y):
        return x + y

    x = np.array([1, 2, 3])
    y = np.array([4, 5, 6])
    print(f"结果：{vec_add(x, y)}")
    \end{lstlisting}
    \end{minted}

    \item 如果电脑没有Python环境，模板同时定义了可选参数\hndm{listType}，如\cref{tab:可选参数说明}所示，通过\hndm{[listType=listings]}，修改抄录环境的宏包为\hndm{listings}，可以用\hndm{listings}宏包排版抄录。模板基于\hndm{listings}宏包定义好了抄录环境样式，通过环境\hndm{lstlisting}使用。模板设置了两种抄录宏包共用一套环境与使用语法，即\hndm{listings}宏包下的\hndm{lstlisting}环境与上文所示的使用方法完全一致。因此可以通过\hndm{[listType=listings]}一键切换宏包。

    \item \hndm{listings}的高亮效果不如\hndm{minted}宏包好，因此模板默认使用\hndm{minted}宏包排版抄录。

    \item 使用\hndm{minted}宏包时，可以通过\hndm{[highlightlines={1,3,5}]}来高亮指定行，多个行号之间使用半角逗号\hndm{,}分隔；模板预设了默认高亮颜色，如果需要指定行高亮颜色，可以通过可选参数\hndm{[highlightcolor=<颜色>]}实现。如果切换到\hndm{listings}宏包，由于\hndm{listings}宏包不支持高亮行操作，因此模板会忽略掉这两个参数。

\end{inparaenum}

\begin{inparaenum}%[]
    \item 模板使用宏包\hndm{biblatex}与命令\hndm{biber}编译参考文献，标准为GB/T 7714-2015，采用顺序编码制。

    \item 需注意，不需要参考文献时，应使用\hndm{XeLaTeX}编译\hndm{.tex}文件；需要使用参考文献时，应使用\hndm{XeLaTeX -> biber -> XeLaTeX -> XeLaTeX}四步编译\hndm{.tex}文件。首次编译时间较长，请耐心等待。如果参考文献文件\hndm{ref.bib}与引用没有更改，可以切换回单次\hndm{XeLaTeX}编译。
